```dockerfile
FROM rustlang/rust:nightly-stretch as BUILD

RUN apt-get update

RUN apt-get install musl-tools -y

RUN rustup target add x86_64-unknown-linux-musl

WORKDIR /usr/simple-api

RUN mkdir src/

COPY . .

RUN RUSTFLAGS=-Clinker=musl-gcc cargo build --release --target=x86_64-unknown-linux-musl

RUN rm -f target/x86_64-unknown-linux-musl/release/deps/simple-api*

# RUN RUSTFLAGS=-Clinker=musl-gcc cargo build --release --target=x86_64-unknown-linux-musl

RUN addgroup --gid 1000 simple-api

RUN adduser --gid 1000 simple-api

RUN chown simple-api:simple-api /usr/simple-api/target/x86_64-unknown-linux-musl/release/simple-api

# ------------------------------------------------------------------------------

FROM gcr.io/distroless/base-debian10

WORKDIR /usr/simple-api/

ENV ROCKET_PORT=8080

COPY --from=BUILD /usr/simple-api/target/x86_64-unknown-linux-musl/release/simple-api .

COPY --from=BUILD /etc/passwd /etc/passwd

COPY --from=BUILD /etc/group /etc/group

USER simple-api

CMD ["./simple-api"]
```
